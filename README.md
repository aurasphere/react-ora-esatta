## Installation

    npm install @aurasphere/react-ora-esatta

or

    yarn add @aurasphere/react-ora-esatta

## Usage

    import { OraEsatta } from '@aurasphere/react-ora-esatta';
    // ...
    if (loading) {
        return <OraEsatta />;
    }
